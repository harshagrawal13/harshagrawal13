\cvsection{Publications}

{\small\textbf{An Energy Landscape Approach to Miniaturizing Enzymes using Protein Language Model Embeddings}}\par
\smallskip
{\footnotesize bioRxiv: \href{https://doi.org/10.64898/2026.03.04.709378}{doi:10.64898/2026.03.04.709378} \hfill March 2026\par\smallskip
\textit{J. L\'{a}la, H. Agrawal, F. Dong, J. Wells, S. Angioletti-Uberti.}\par\smallskip
\begin{itemize}
    \item {Developed an AI-guided framework to miniaturize enzymes by sampling a protein language model (PLM) embedding energy landscape via Monte Carlo optimization (BAGEL), identifying maximally compact sequences that preserve catalytic site geometry --- without requiring any structural information.}
    \item {Validated designs across four enzymes (PETase, subtilisin Carlsberg, Taq DNA polymerase, VioA) using a two-stage pipeline of deep-learning structure predictors (ESMFold, Chai-1, Boltz-2) and molecular dynamics simulations; top candidates achieved active-site RMSD $<$ 2 \AA{} and RMSF $<$ 1.0 \AA{} relative to wild-type. Open-sourced all generated sequences to accelerate experimental validation. \textit{Experimental validation underway.}}
\end{itemize}}
